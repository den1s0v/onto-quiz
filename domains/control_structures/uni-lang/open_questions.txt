# 2. "Почему эта строка следующая?"

# алгоритм № 1

функция main
{
	если усл_1 разв_1
	{
		ждать
	}
	иначе если усл_2 
	{
		бежать
	}
	иначе
	{
		лежать
	}
}


# Правильная трасса

началась программа
началась функция main
	началась разв_1
		выполнилось усл_1 - ложь		# <- спросить про эту строку (проверяется первое условие в цепочке if-else-if)
		выполнилось усл_2 - истина		# <- спросить про эту строку (проверяется следующее условие, т.к. предыдущее ложно)
		началась ветка 2 разв_1		# <- спросить про эту строку (началась ветка 2 развилки 1, т.к. соответвующее ему условие 2 истинно)
			выполнилось бежать		# <- спросить про эту строку (действие "бежать" входит в ветку 2 развилки 1)
		закончилась ветка 2 разв_1
	закончилась разв_1		# <- спросить про эту строку (развилка завершена, т.к. выполнилась ветка по истинному условию)
закончилась функция main		# <- спросить про эту строку (функция main завершилась, т.к. развилка 1 была последним (и единственным) действием в теле функции main)
закончилась программа
